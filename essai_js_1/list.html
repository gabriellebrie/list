function state(e) { // body... var el; if (!e) { e = window.event; } el = e.target || e.srcElement; if (el.className === 'hot') { el.classList.replace('hot', 'normal'); } else if (el.className === 'normal') { el.classList.replace('normal', 'cool'); } else if (el.className === 'cool') { el.classList.replace('cool', 'completed'); } else { el.classList.replace('completed', 'hot') } } var elClick = document.getElementsByClassName('list') if (elClick.addEventListener) { elClick.addEventListener('click', function(e) {state(e)}, false); }
